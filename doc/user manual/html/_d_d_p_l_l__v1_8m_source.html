131  function [E_hat, phaseEstimate] = pll(obj, Ein, Ts, Kv, tau1, tau2, refConstellation, initialPhase)
132  N = length(Ein);
133 
134  % Loop filter coefficients
135  a1b = [1 Ts/(2*tau1) * ( 1+[-1 1]/tan(Ts/(2*tau2)) ) ];
136 
137  u_d = 0; % Output of phase detector (residual phase error)
138  u_f = 0; % Output of loop filter
139  phaseEstimate = initialPhase*ones(N+1,1);
140  E_hat = zeros(N,1);
141 
142  for n=1:N
143  u_d1 = u_d;
144  E_hat(n) = Ein(n)*exp(-1j*phaseEstimate(n)); % Remove estimate of phase erro from input symbol
145  [~,idx] = min(abs(refConstellation-E_hat(n))); % Slicer (perform hard decision on symbol)
146  u_d = imag(E_hat(n)*conj(refConstellation(idx))); % Generate phase error signal (also called x_n (Meyer))
147  u_f = sum(a1b.*[u_f u_d1 u_d]); % Process phase error signal in Loop Filter (also called e_n (Meyer))
148  phaseEstimate(n+1) = phaseEstimate(n) + Kv*u_f; % Estimate the phase error for the next symbol
149  end
150  phaseEstimate = phaseEstimate(1:end-1); % Remove the last estimate, which is unuseful
151  end
